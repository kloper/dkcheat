\noalign{\bf{Sums of Trigonometric Functions} \hfil \vskip 1mm} \cr
\noalign{\hrule\vskip 1 mm}
\sin\alpha + \sin\beta = 2\sin {{\alpha + \beta} \over 2} %
\cos {{\alpha - \beta} \over 2} \cr
\noalign{\vskip 1ex}
%
\cos\alpha + \cos\beta = 2\cos {{\alpha + \beta} \over 2} %
\cos {{\alpha - \beta} \over 2} \cr
\noalign{\vskip 1ex}
%
\tan\alpha + \tan\beta = {{\sin(\alpha + \beta)} \over %
{\cos\alpha\cos\beta}}\cr
\noalign{\vskip 1ex}
%
\cot\alpha + \cot\beta = {{\sin(\alpha + \beta)} \over %
{\sin\alpha\sin\beta}} \cr
\noalign{\vskip 1ex}
%
\sin\alpha - \sin\beta = 2\sin {{\alpha + \beta} \over 2} %
\sin {{\alpha - \beta} \over 2} \cr
\noalign{\vskip 1ex}
%
\cos\alpha - \cos\beta = -2\sin {{\alpha + \beta} \over 2} %
\sin {{\alpha - \beta} \over 2} \cr
\noalign{\vskip 1ex}
%
\tan\alpha - \tan\beta = {{\sin(\alpha - \beta)} \over %
{\cos\alpha\cos\beta}}\cr
\noalign{\vskip 1ex}
%
\cot\alpha - \cot\beta = {{\sin(\beta - \alpha)} \over %
{\sin\alpha\sin\beta}} \cr
\noalign{\vskip 1ex}
%
\cos\alpha + \sin\alpha = \sqrt{2}\cos({\pi \over 4} - \alpha) \cr
\noalign{\vskip 1ex}
%
\cos\alpha - \sin\alpha = \sqrt{2}\sin({\pi \over 4} - \alpha) \cr
\noalign{\vskip 1ex}
%
\cos\alpha\cos\beta = {1 \over 2}(\cos(\alpha - \beta) + \cos(\alpha + \beta))
\cr
\noalign{\vskip 1ex}
%
\sin\alpha\cos\beta = {1 \over 2}(\sin(\alpha - \beta) + \sin(\alpha + \beta))
\cr
\noalign{\vskip 1ex}
%
\sin\alpha\sin\beta = {1 \over 2}(\cos(\alpha - \beta) - \cos(\alpha + \beta))
\cr
