\noalign{\bf{Properties of Powers} \hfil \vskip 1mm} \cr
\noalign{\hrule\vskip 1 mm}
\displaystyle a^ma^n=a^{m+n}, \hskip 3em %
\displaystyle {a^m \over a^n} = a^{m-n}, \hskip 3em %
\displaystyle {1 \over a^n} = a^{-n} \cr\noalign{\vskip 1mm}
\displaystyle (a^m)^n = a^{mn}, \hskip 3em %
\displaystyle \root n \of a^m = a^{m \over n} \cr

\noalign{\bf{Properties of Logarithms} \hfil \vskip 1mm} \cr
\noalign{\hrule\vskip 1 mm}
\displaystyle a^n \Rightarrow \log_a n \cr
\displaystyle \log_a a = 1, \hskip 2em %
\displaystyle \log_a 1 = 0, \hskip 2em %
\displaystyle a^{log_a x} = x \hskip 2em \cr
\noalign{\vskip 1mm}
\displaystyle \log_a a^x = x, \hskip 2em %
\displaystyle \log_a {1 \over x} = -\log_a x, \hskip 2em %
\displaystyle \log_a x^n = n\log_a x \cr
\displaystyle \log_{a^n} x = {1 \over n}\log_a x, \hskip 2em %
\displaystyle \log_b x = {{\log_a x} \over {\log_a b}} \cr
\noalign{\vskip 1mm}
\displaystyle \log_a xy = {\log_a x} + {\log_a y}, \hskip 2em %
\displaystyle \log_a {x \over y} = {\log_a x} - {\log_a y} \cr
