\noalign{\bf{Double-Angle} \hfil} \cr
\noalign{\hrule\vskip 1 mm}
\sin 2\alpha = 2\sin\alpha\cos\alpha &
\cos 2\alpha = \cos^2\alpha - \sin^2\alpha \cr
\noalign{\vskip 1ex}
%
\cos 2\alpha = 1 - 2\sin^2\alpha &
\cos 2\alpha = 2\cos^2\alpha - 1 \cr
\noalign{\vskip 1ex}
%
\tan 2\alpha = {{2\tan\alpha} \over {1 - \tan^2\alpha}} &
\cot 2\alpha = {{\cot^2\alpha - 1} \over {2\cot\alpha}} \cr
\noalign{\vskip 1ex}
%
\sin 2\alpha = {{2\tan\alpha} \over {1 + \tan^2\alpha}} &
\cos 2\alpha = {{1 - \tan^2\alpha} \over {1 + \tan^2\alpha}} \cr
\noalign{\vskip 1ex}
%
\tan\alpha + \cot\alpha = {2 \over {\sin 2\alpha}} &
\cot\alpha - \tan\alpha = {2 \over {\tan 2\alpha}} \cr
\noalign{\vskip 1ex}
%
1 - \tan^2\alpha = {{\cos 2\alpha} \over {\cos^2\alpha}} &
1 - \cot^2\alpha = -{{\cos 2\alpha} \over {\sin^2\alpha}} \cr
\noalign{\vskip 1ex}
%
\sin 3\alpha = 3\sin\alpha - 4\sin^3\alpha &
\cos 3\alpha = 4\cos^3\alpha - 3\cos\alpha \cr
