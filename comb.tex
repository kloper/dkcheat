\noalign{\bf{Combinatorics} \hfil \vskip 1mm} \cr
\noalign{\hrule\vskip 1 mm}
\displaystyle {n \choose k} = {n! \over (n-k)! k!} \cr
\noalign{\vskip 1 mm}
\displaystyle {n \choose k} = {n \choose n-k} \cr
\noalign{\vskip 1 mm}
\displaystyle {n \choose k} = {n\over k}{n-1 \choose k-1} \cr
\noalign{\vskip 1 mm}
\displaystyle {n \choose k} = {n -1 \choose k} + {n-1 \choose k-1} \cr
\noalign{\vskip 1 mm}
\displaystyle {n \choose m}{m \choose k} = { n \choose k} { n-k \choose m-k}\cr
\noalign{\vskip 1 mm}
\displaystyle {n \choose k} =  (-1)^k {k-n-1 \choose k} \cr
\noalign{\vskip 2 mm}
\displaystyle \sum_{k=0}^n {n \choose k} = 2^n \cr
\noalign{\vskip 1 mm}
\displaystyle \sum_{k=0}^n {r+k \choose k} = { r+ n+1 \choose n} \cr
\noalign{\vskip 1 mm}
\displaystyle \sum_{k=0}^n {k \choose m} = { n+1 \choose m+1} \cr
\noalign{\vskip 1 mm}
\displaystyle \sum_{k=0}^n {r \choose k}{s \choose n -k} = {r+s \choose n} \cr
