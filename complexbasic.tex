\def\za{{z_{1}}}
\def\zb{{z_{2}}}
\def\bza{{\bar{z}_{1}}}
\def\bzb{{\bar{z}_{2}}}
\def\ra{{r_{1}}}
\def\rb{{r_{2}}}
\def\ta{{\theta_{1}}}
\def\tb{{\theta_{2}}}
\halign{\tabskip=1em plus 4em \quad%

$#\hfil$%
\cr
\noalign{\bf{Complex Numbers} \hfil} \cr
\noalign{\hrule\vskip 1 mm}
z = (x, y) = x + iy & x, y \in R, \hskip 1ex i^2 = -1 \cr
\noalign{\vskip 1ex}
(0,0) \equiv 0 \in C & (1, 0) \equiv 1 \in C \cr
\noalign{\vskip 1ex}
(x, y) + (u, v) = (x+u, y+v) \span \cr
\noalign{\vskip 1ex}
(x, y)(u, v) = (xu-yv, xv+yu) \span \cr
\noalign{\vskip 1em} 
\noalign{\bf{Absolute Value} \hfil \vskip 0.3ex} \cr
\noalign{\hrule\vskip 1 mm}
|z| = \sqrt{x^2+y^2} \span  \cr
\noalign{\vskip 1ex}
|z| \geq 0 & |z| = 0 \hskip 1ex \Leftrightarrow \hskip 1ex z = 0 \cr
\noalign{\vskip 1ex}
|-z| = |z| & |\za - \zb| = |\zb - \za| \cr
\noalign{\vskip 1ex}
|\za\zb| = |\za||\zb| &  |{{\za} \over {\zb}}| = {{|\za|} \over {|\zb|}} \cr
\noalign{\vskip 1ex}
|\za + \zb| \leq |\za| + |\zb| \hskip 1ex (triangle~inequality) \span \cr
\noalign{\vskip 1ex}
|\za - \zb| \leq |\za| + |\zb| \span \cr
\noalign{\vskip 1ex}
||\za| - |\zb|| \leq |\za - \zb| \span \cr
\noalign{\vskip 1em} 
\noalign{\bf{Polar Form} \hfil \vskip 0.3ex} \cr
\noalign{\hrule\vskip 1 mm}
e^{i\theta} = \cos\theta + i\sin\theta \hskip 1em (Euler's~formula) \span \cr
\noalign{\vskip 1ex}
z = r\cos\theta + r\sin\theta = re^{i\theta} & r = |z| = \sqrt{x^2 + y^2} \cr
\noalign{\vskip 1ex}
& \cos\theta = {x \over \sqrt{x^2+y^2}} \cr
& \sin\theta = {y \over \sqrt{x^2+y^2}} \cr
\noalign{\vskip 1ex}
e^{i\theta} = 1 \hskip 1ex \Leftrightarrow \hskip 1ex \theta = 2\pi n & %
e^0 = 1 \cr
\noalign{\vskip 1ex}
e^{i(\theta+2\pi)} = e^{i\theta}, \hskip 1ex periodic \hskip 1ex T = 2\pi %
\span\cr
\noalign{\vskip 1ex}
(e^{i\theta})^n = e^{i\theta n}, \hskip 1ex n \in Z %
\hskip 1ex (DeMoire's~theorem) \span \cr
\noalign{\vskip 1ex}
\za\zb=\ra\rb~e^{i(\ta+\tb)} & %
{\za \over \zb} = {\ra \over \rb} e^{i(\ta - \tb)} \cr
\noalign{\vskip 1em} 
\noalign{\bf{Conjugate} \hfil} \cr
\noalign{\hrule\vskip 1 mm}
\bar{z} = \overline{x + iy} = x - iy \cr
\noalign{\vskip 1ex}
z = \bar{z} \hskip 1ex \Leftrightarrow \hskip 1ex z \in R & %
z + \bar{z} = 2x \in R \cr
\noalign{\vskip 1ex}
\overline{\za + \zb} = \bza + \bzb &
\overline{\za \zb} = \bza\bzb \cr
\noalign{\vskip 1ex}
z\bar{z} = {|z|}^2 \in R & %
\bar{z} = {1 \over z} \hskip 1ex \Leftrightarrow \hskip 1ex |z| = 1 \cr
\noalign{\vskip 1ex}
\bar{z} = re^{-i\theta} \cr
\noalign{\vskip 1em} 
\noalign{\bf{Roots} \hfil \vskip 0.3ex} \cr
\noalign{\hrule\vskip 1 mm}
{(re^{i\theta})}^n = Re^{i\alpha}, \hskip 1ex n \in Z %
\hskip 1ex & \Leftrightarrow \hskip 1ex 
\cases {
r = \root n \of R \cr
e^{i\theta n} = e^{i\alpha} \cr
} \cr
\noalign{\vskip 1ex} 
\theta = {\alpha \over n} + \phi & %
\phi \in \{ 0, {{2\pi} \over n}, %
{{4\pi} \over n}, ..., {{2n-2} \over n}\}\cr
